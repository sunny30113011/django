6Ô∏è‚É£ Student Model

# student/models.py

from django.db import models

class Student(models.Model):
    name = models.CharField(max_length=100)
    age = models.IntegerField()
    email = models.EmailField()

    def __str__(self):
        return self.name

        fields = ['name', 'age', 'email']

# student/views.py

def stud_delete(requset,pk):
        studentdata=get_object_or_404(Student,pk=pk)
        if requset.method =='POST':
            studentdata.delete()  
            return redirect('details')
        return render(requset,'formapp/delete.html',{'form':studentdata})

9Ô∏è‚É£ update View in views.py

# student/views.py

def stud_update(request,pk):
    studentdata=get_object_or_404(Student,pk=pk)
    form=StudentForm(request.POST or None,instance=studentdata)
    if form.is_valid():
        form.save()
        return redirect('details')

    return render(request,'formapp/index.html',{'form':form})

9Ô∏è‚É£ search View in views.py

# student/views.py

def searchdata(request):
     if request.method=='POST':
          data=request.POST["search"]
          namedata=Student.objects.filter(name__icontains=data)
          agedata=Student.objects.filter(age__icontains=data)
          emaildata=Student.objects.filter(email__icontains=data)
          alldata=list(set(list(namedata)+list(agedata)+list(emaildata)))

     return render(request,'formapp/search.html',{'form':alldata})
